def cd(path)
  if path == '..'
    self.parent_context
  else
    item = item_at(path)
    if item.nil?
      nil
    else
      Context.new(item,self)
    end
  end
end

def to_s
  if self.here.kind_of? Array
    indices = []
    self.here.each_index { |i| indices << i }
    indices.join(' ')
  elsif self.here.kind_of? Hash
    self.here.keys.join(' ')
  else
    self.here.to_s
  end
end
